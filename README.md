#My personal vimrc

Feel free to use it / modify it / do whatever with it.
Install instructions:

	git clone https://github.com/dapetcu21/vimrc.git
	cd vimrc
	git submodule init
	git submodule update
	ln -s .vimrc ~/.vimrc
	ln -s .vim ~/.vim
	vim
	:BundleInstall
	#restart vim
	:SetUpPlugins
	#restart vim
